## 一、测试实施概述

### 1.1 实施目标

**核心目标**: 建立基于五大智能引擎+一个AI训练中心的严格测试体系，确保智能分析系统能够解决真实问题，而不是绕过问题

**具体目标**:
- 验证DataPipeline（数据管道）的完整性和准确性
- 验证QuantSignalEngine（量化信号引擎）的计算正确性
- 验证HotspotIntelligenceEngine（热点智能引擎）的算法准确性
- 验证AttributionEngine（归因引擎）的分析深度
- 验证PredictionEngine（预测引擎）的预测准确性
- 验证LearningLoopCoordinator（学习闭环协调器）的协调能力
- 验证AI Training Center（AI训练中心）的训练效果
- 按照正确的业务顺序进行测试
- 验证引擎间协作的正确性

### 1.2 实施原则

**问题解决导向**:
- ❌ 禁止使用模拟数据掩盖问题
- ❌ 禁止简化测试场景避免问题
- ❌ 禁止使用固定数据避免变化
- ✅ 必须使用真实数据测试
- ✅ 必须测试边界条件和异常情况
- ✅ 必须验证系统在真实环境下的表现

## 三、测试环境配置

### 3.1 容器化部署配置

#### 3.1.1 Docker Compose配置
**核心思想**: 使用Docker Compose一键启动所有服务，服务间依赖关系由`depends_on`管理

**docker-compose.yml配置**:
```yaml
version: '3.8'
services:
  # 数据管道服务
  datapipeline:
    build: ./docker/datapipeline
    environment:
      - NODE_ENV=test
      - TUSHARE_TOKEN=${TUSHARE_TOKEN}
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
    depends_on:
      - redis
      - postgres
    volumes:
      - ./data:/app/data
      - ./logs:/app/logs

  # 量化信号引擎服务
  quantsignal:
    build: ./docker/quantsignal
    environment:
      - NODE_ENV=test
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
    depends_on:
      - datapipeline
      - redis
      - postgres

  # 热点智能引擎服务
  hotspot:
    build: ./docker/hotspot
    environment:
      - NODE_ENV=test
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
    depends_on:
      - quantsignal
      - redis
      - postgres

  # 归因引擎服务
  attribution:
    build: ./docker/attribution
    environment:
      - NODE_ENV=test
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
      - LLM_API_KEY=${LLM_API_KEY}
    depends_on:
      - hotspot
      - redis
      - postgres

  # 预测引擎服务
  prediction:
    build: ./docker/prediction
    environment:
      - NODE_ENV=test
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
    depends_on:
      - attribution
      - redis
      - postgres

  # 学习闭环协调器服务
  learning:
    build: ./docker/learning
    environment:
      - NODE_ENV=test
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
    depends_on:
      - prediction
      - redis
      - postgres

  # AI训练中心服务
  training:
    build: ./docker/training
    environment:
      - NODE_ENV=test
      - REDIS_URL=redis://redis:6379
      - POSTGRES_URL=postgresql://postgres:password@postgres:5432/test_db
    depends_on:
      - learning
      - redis
      - postgres

  # 基础设施服务
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

  postgres:
    image: postgres:15-alpine
    environment:
      - POSTGRES_DB=test_db
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=password
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data

volumes:
  redis_data:
  postgres_data:
```

export class RealDataLoader {
  async loadNewsData(period: DateRange): Promise<NewsItem[]> {
    const newsData: NewsItem[] = [];
    
    // 从多个真实数据源加载
    for (const [sourceName, source] of Object.entries(RealDataSources.NEWS_SOURCES)) {
      try {
        const data = await this.fetchFromSource(source, period);
        newsData.push(...data);
        
        // 验证数据质量
        this.validateNewsData(data, sourceName);
      } catch (error) {
        console.error(`Failed to load data from ${sourceName}:`, error);
        // 继续从其他源加载，不中断测试
      }
    }
    
    // 去重和排序
    const uniqueNews = this.deduplicateNews(newsData);
    const sortedNews = this.sortNewsByTime(uniqueNews);
    
    return sortedNews;
  }
  
  private validateNewsData(data: NewsItem[], source: string): void {
    for (const news of data) {
      // 验证必需字段
      if (!news.title || news.title.length < 10) {
        throw new Error(`Invalid title from ${source}: ${news.title}`);
      }
      
      if (!news.content || news.content.length < 100) {
        throw new Error(`Invalid content from ${source}: ${news.content}`);
      }
      
      // 验证时间格式
      if (!this.isValidDate(news.publishTime)) {
        throw new Error(`Invalid publish time from ${source}: ${news.publishTime}`);
      }
    }
  }
}

export class DataQualityValidator {
  validateFourLayerData(data: FourLayerMarketData[]): ValidationResult {
    const result: ValidationResult = {
      isValid: true,
      errors: [],
      warnings: []
    };
    
    for (const marketData of data) {
      // 验证第1层：宽基指数层
      if (!marketData.layer1BroadIndices || marketData.layer1BroadIndices.length !== 7) {
        result.errors.push(`Invalid layer1 data: expected 7 broad indices, got ${marketData.layer1BroadIndices?.length || 0}`);
        result.isValid = false;
      }
      
      // 验证第2层：一级指数层
      if (!marketData.layer2PrimaryIndices || marketData.layer2PrimaryIndices.length !== 17) {
        result.errors.push(`Invalid layer2 data: expected 17 primary indices, got ${marketData.layer2PrimaryIndices?.length || 0}`);
        result.isValid = false;
      }
      
      // 验证第3层：二级指数层
      if (!marketData.layer3SecondaryIndices || marketData.layer3SecondaryIndices.length !== 43) {
        result.errors.push(`Invalid layer3 data: expected 43 secondary indices, got ${marketData.layer3SecondaryIndices?.length || 0}`);
        result.isValid = false;
      }
      
      // 验证第4层：龙头股层
      if (!marketData.layer4LeadingStocks || marketData.layer4LeadingStocks.length !== 194) {
        result.errors.push(`Invalid layer4 data: expected 194 leading stocks, got ${marketData.layer4LeadingStocks?.length || 0}`);
        result.isValid = false;
      }
      
      // 验证日期
      if (!this.isValidDate(marketData.date)) {
        result.errors.push(`Invalid date: ${marketData.date}`);
        result.isValid = false;
      }
      
      // 验证数据完整性
      if (!this.validateDataIntegrity(marketData)) {
        result.warnings.push(`Data integrity warning for date: ${marketData.date}`);
      }
    }
    
    return result;
  }
  
  validateQuantSignals(data: QuantSignalData[]): ValidationResult {
    const result: ValidationResult = {
      isValid: true,
      errors: [],
      warnings: []
    };
    
    for (const signal of data) {
      // 验证宏观风险偏好Z分数
      if (typeof signal.macroRiskZScore !== 'number' || isNaN(signal.macroRiskZScore)) {
        result.errors.push(`Invalid macro risk Z score: ${signal.macroRiskZScore}`);
        result.isValid = false;
      }
      
      // 验证市场风格Z分数
      if (!signal.styleZScores || signal.styleZScores.length !== 5) {
        result.errors.push(`Invalid style Z scores: expected 5, got ${signal.styleZScores?.length || 0}`);
        result.isValid = false;
      }
      
      // 验证量化指纹Z分数
      if (typeof signal.quantFingerprintZScore !== 'number' || isNaN(signal.quantFingerprintZScore)) {
        result.errors.push(`Invalid quant fingerprint Z score: ${signal.quantFingerprintZScore}`);
        result.isValid = false;
      }
    }
    
    return result;
  }
  
  private validateDataIntegrity(marketData: FourLayerMarketData): boolean {
    // 检查数据完整性
    return marketData.layer1BroadIndices.every(item => item.code && item.name) &&
           marketData.layer2PrimaryIndices.every(item => item.code && item.name) &&
           marketData.layer3SecondaryIndices.every(item => item.code && item.name) &&
           marketData.layer4LeadingStocks.every(item => item.code && item.name);
  }
